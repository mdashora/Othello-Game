create table "OTHELLO".USERS
(
	ID INTEGER default GENERATED_BY_DEFAULT not null primary key,
	NAME VARCHAR(255),
	PWD VARCHAR(255),
	SALT VARCHAR(255),
	UUID VARCHAR(255),
	GAMESWON INTEGER default 0,
	GAMESLOST INTEGER default 0
)

create table "OTHELLO".ROLES
(
	ID INTEGER default GENERATED_BY_DEFAULT not null primary key,
	NAME VARCHAR(255),
	USERROLE VARCHAR(255),
	USERID INTEGER
)

create table "OTHELLO".GAMES
(
	ID INTEGER default GENERATED_BY_DEFAULT not null primary key,
	NAME VARCHAR(255),
	WHITEPLAYER VARCHAR(255),
	BLACKPLAYER VARCHAR(255),
	GAMESTATUS VARCHAR(255) not null,
	GAMEMOVES VARCHAR(255) not null,
	
)
